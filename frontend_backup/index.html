kind: TopicBatch
env: prod
change_id: "2025-10-02_001"
items:
  - name: prod.orders.created
    action: create
    config:
      partitions: 12
      replication_factor: 3
      retention_ms: 604800000
    metadata:
      owner: team-commerce
      doc: "https://wiki.example.com"
      tags: ["orders", "events"]
